def binary_Search(a, N, key):
    start = 0
    end = N-1
    while start <= end:
        mid = (start + end)//2
        if a[mid] < key:
            start = mid + 1
        elif a[mid] > key:
            end = mid - 1
        else :
            return True
    if key not in a:
        return False
